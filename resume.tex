%% CATEGORIES

\newcommand{\sect}[1]% use for a new level 1 category
	{
	\section{{#1}}
	\hrule
	}

\newcommand{\sectlist}[1]% use for a new level 1 category that goes straight to itemize
	{
	\section{{#1}}
	\hrule
	\vspace{\GapTwo}% gap to account for \hrule
	}

\newcommand{\school}[4]% use for a level 2 item that is a school; inputs are school name, major, graduation date, GPA
	{
	\vspace{\GapTwo}% gap to account for \hrule
	\begin{minipage}{0.5\textwidth}
		\begin{flushleft}
			\textbf{{#1}}% school name
		\end{flushleft}
	\end{minipage}
	\hfill
	\begin{minipage}{0.5\textwidth}
		\begin{flushright}
			{#3}% graduation date
		\end{flushright}
	\end{minipage}

	\begin{minipage}{0.5\textwidth}
		\begin{flushleft}
			{#2}% major
		\end{flushleft}
	\end{minipage}
	\hfill
	\begin{minipage}{0.5\textwidth}
		\begin{flushright}
			{#4}% GPA
		\end{flushright}
	\end{minipage}
	}

\newcommand{\entry}[4]% use for a level 2 item; inputs are name, position, location, dates
	{
	\vspace{\GapTwo}% gap to account for \hrule
	\begin{minipage}{0.5\textwidth}
		\begin{flushleft}
			\textbf{{#1}}% name
		\end{flushleft}
	\end{minipage}
	\hfill
	\begin{minipage}{0.5\textwidth}
		\begin{flushright}
			{#3}% location
		\end{flushright}
	\end{minipage}

	\begin{minipage}{0.5\textwidth}
		\begin{flushleft}
			\textit{#2}% position
		\end{flushleft}
	\end{minipage}
	\hfill
	\begin{minipage}{0.5\textwidth}
		\begin{flushright}
			{#4}% dates
		\end{flushright}
	\end{minipage}
	}
