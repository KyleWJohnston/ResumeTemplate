%% CATEGORIES

\newcommand{\sect}[1]% use for a new level 1 category
	{
	\section{{#1}}
	\hrule
	}

\newcommand{\sectlist}[1]% use for a new level 1 category that goes straight to itemize
	{
	\section{{#1}}
	\hrule
	\vspace{\GapTwo}% gap to account for \hrule (one column list only)
	}

\newcommand{\school}[4]% use for a level 2 item that is a school; inputs are school name, major, graduation date, GPA
	{
	\vspace{\GapTwo}% gap to account for \hrule
	\textbf{{#1}} \hfill {#3}\\% school and graduation date
	{#2} \hfill {#4}% major and GPA
	}

\newcommand{\entry}[4]% use for a level 2 item; inputs are company, position, location, dates
	{
	\vspace{\GapTwo}% gap to account for \hrule
	\textbf{{#1}}\\% company
	\textit{#2} \textit{(#3)} \hfill {#4}% position, location, dates
	}

\newcommand{\role}[3]% use for continuing a level 2 item; inputs are position, location, dates
	{
	\vspace{\GapTwo}% gap to account for \hrule
	\textit{#1} \textit{(#2)} \hfill {#3}% position, location, dates
	}
